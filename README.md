├── deployment

│   └── flaskr.conf
├── flaskrapp

│   │   └── style.css
│   ├── templates
│   │   ├── layout.html
│   │   ├── login.html
│   │   └── show_entries.html
│   └── views.py
├── README.md
├── requirements.txt
├── run.py
└── schema.sql


2. pip install flask gunicorn
